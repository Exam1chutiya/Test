def count_ones(matrix):
rows = len(matrix)
cols = len(matrix[0]) if rows > 0 else 0

# Initialize row and column counts
row_counts = [0] * rows
col_counts = [0] * cols

# Iterate through the matrix
for i in range(rows):
for j in range(cols):
if matrix[i][j] == 1:
row_counts[i] += 1
col_counts[j] += 1

# Output row-wise and column-wise counts
print("Row-wise counts:", row_counts)
print("Column-wise counts:", col_counts)

# Example usage
matrix = [
[0, 1, 0],
[1, 1, 1],
[0, 0, 1]
]
count_ones(matrix)
